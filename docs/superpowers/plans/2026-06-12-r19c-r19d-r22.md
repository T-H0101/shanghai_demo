// lib/site-agent/command-queue.ts
/**
 * Site Agent Command Queue (SQLite)
 *
 * 用途:
 *   - dedupe by commandId (Agent 拉取幂等)
 *   - 状态机: pending -> pulled -> {success | failed | unsupported}
 *   - Agent 重启可恢复 (本地持久化)
 *   - 后续 Sync Pusher 和 Ctrl Poller 共享
 *
 * R.18 严口径:
 *   - 仅本地持久化, 不写总控库
 *   - 仅记录执行状态, 不代替 control_command.result
 */

import Database from "better-sqlite3"
import { dirname } from "node:path"
import { mkdirSync } from "node:fs"

export type CommandState = "pending" | "pulled" | "success" | "failed" | "unsupported"

export interface PendingCommand {
  commandId: string
  commandType: string
  payload: string
  createdAt: string
}

export class CommandQueue {
  private db: Database.Database

  constructor(dbPath: string) {
    mkdirSync(dirname(dbPath), { recursive: true })
    this.db = new Database(dbPath)
    this.db.pragma("journal_mode = WAL")
    this.db.exec(`
      CREATE TABLE IF NOT EXISTS command_queue (
        command_id TEXT PRIMARY KEY,
        command_type TEXT NOT NULL,
        payload TEXT NOT NULL,
        state TEXT NOT NULL DEFAULT 'pending',
        result TEXT,
        error_message TEXT,
        created_at TEXT NOT NULL DEFAULT (datetime('now')),
        updated_at TEXT NOT NULL DEFAULT (datetime('now'))
      )
    `)
  }

  /**
   * 标记新命令. 已存在返 false (dedupe).
   */
  markPending(commandId: string, commandType: string, payload: string): boolean {
    const stmt = this.db.prepare(
      `INSERT OR IGNORE INTO command_queue (command_id, command_type, payload)
       VALUES (?, ?, ?)`
    )
    const r = stmt.run(commandId, commandType, payload)
    return r.changes > 0
  }

  markPulled(commandId: string): CommandState {
    const stmt = this.db.prepare(
      `UPDATE command_queue SET state = 'pulled', updated_at = datetime('now')
       WHERE command_id = ? AND state = 'pending'`
    )
    const r = stmt.run(commandId)
    if (r.changes === 0) return this.getState(commandId)
    return "pulled"
  }

  markResult(commandId: string, status: "success" | "failed" | "unsupported", result: string, errorMessage: string | null): CommandState {
    const stmt = this.db.prepare(
      `UPDATE command_queue SET state = ?, result = ?, error_message = ?, updated_at = datetime('now')
       WHERE command_id = ?`
    )
    stmt.run(status, result, errorMessage, commandId)
    return this.getState(commandId)
  }

  getState(commandId: string): CommandState {
    const row = this.db
      .prepare(`SELECT state FROM command_queue WHERE command_id = ?`)
      .get(commandId) as { state: CommandState } | undefined
    return row?.state ?? "pending" // 不存在视为 pending
  }

  listPending(): PendingCommand[] {
    return this.db
      .prepare(
        `SELECT command_id AS commandId, command_type AS commandType, payload, created_at AS createdAt
         FROM command_queue WHERE state = 'pending' ORDER BY created_at ASC`
      )
      .all() as PendingCommand[]
  }

  close(): void {
    this.db.close()
  }
}
```

// lib/site-agent/pg-reader.ts
/**
 * Site Agent PG Reader (R.19C)
 *
 * 用途: 增量读 source_restore 6 张白名单, 输出 unified_* 字段名 package rows.
 *
 * R.18 严口径:
 *   - 只读白名单 6 张 (R.18 integration matrix 认证)
 *   - 增量靠 update_dt / create_dt
 *   - 不直连中心库 (那是 inlineUpsert 走总控)
 */

import { Pool, type PoolClient } from "pg"

export const SOURCE_TABLES = [
  "tbl_task",
  "tbl_disc",
  "tbl_logical_volume",
  "tbl_magzines",
  "tbl_hd_info",
  "tbl_disc_lib",
] as const

export type SourceTableName = (typeof SOURCE_TABLES)[number]

export interface SourcePackageRow {
  source_table: SourceTableName
  source_id: string
  [key: string]: unknown
}

const TABLE_WATERMARK_COLUMN: Record<SourceTableName, string> = {
  tbl_task: "update_dt",
  tbl_disc: "update_dt",
  tbl_logical_volume: "update_time",
  tbl_magzines: "earliest_time", // 实际无 update_dt, 用 earliest_time
  tbl_hd_info: "create_dt",
  tbl_disc_lib: "device_order",  // tbl_disc_lib 无 update_dt, 走全量 (R.18 已知)
}

export function mapSourceRowToUnified(tableName: SourceTableName, row: Record<string, unknown>): SourcePackageRow {
  const out: SourcePackageRow = {
    source_table: tableName,
    source_id: String(row.id ?? ""),
  }
  // 透传所有字段 (中心端 inlineUpsert 按白名单列映射)
  for (const [k, v] of Object.entries(row)) {
    if (k === "id") continue
    out[k] = v
  }
  return out
}

export interface PgReaderConfig {
  siteDatabaseUrl: string
}

export class PgReader {
  private pool: Pool

  constructor(config: PgReaderConfig) {
    this.pool = new Pool({ connectionString: config.siteDatabaseUrl, max: 3, idleTimeoutMillis: 10_000 })
  }

  /**
   * 增量读 1 张表, 返 rows.
   *  - R.19C: 透传所有字段, 中心端按白名单 inlineUpsert.
   *  - watermark: 上次同步时间 (ISO string).
   *  - 缺 watermark 列时, 走 id>0 全量 (R.18 已知: tbl_disc_lib / tbl_magzines).
   */
  async readIncremental(
    table: SourceTableName,
    watermark: string | null,
    limit: number = 500
  ): Promise<Record<string, unknown>[]> {
    const col = TABLE_WATERMARK_COLUMN[table]
    const client = await this.pool.connect()
    try {
      let result
      if (watermark && col !== "device_order") {
        result = await client.query(
          `SELECT * FROM ${table} WHERE ${col} > $1::timestamptz ORDER BY ${col} ASC LIMIT $2`,
          [watermark, limit]
        )
      } else {
        result = await client.query(`SELECT * FROM ${table} ORDER BY id ASC LIMIT $1`, [limit])
      }
      return result.rows
    } finally {
      client.release()
    }
  }

  async close(): Promise<void> {
    await this.pool.end()
  }
}

// app/api/site-agent/packages/route.ts
/**
 * POST /api/site-agent/packages (R.19C)
 *
 * Site Agent 推小表 package 入口.
 * R.19A heartbeat 已建, R.19C 加 push.
 *
 * 复用 R.2H.2 package-dispatcher (R.18 协议冻结).
 *
 * 行为:
 *   - 验签 (verifySiteAgentRequest, R.19A 已有)
 *   - 验 schema (白名单 6 张: R.19C)
 *   - 调 dispatchPackage (R.2H.2 已有)
 *   - 写 sync_package_log + audit_log
 *   - 返 {batchId, accepted, rejected, errors}
 *
 * 边界 (R.18):
 *   - 仅接受白名单表 (R.18 integration matrix)
 *   - 不替代 R.2H.2 dispatcher
 *   - Agent 仿真不等于真站点 app 完成
 */

import { NextRequest, NextResponse } from "next/server"
import { verifySiteAgentRequest } from "@/lib/site-agent/hmac"
import { transaction, query } from "@/lib/db"
import { dispatchPackage } from "@/lib/sync/package-dispatcher"
import { ALLOWED_PACKAGE_TABLES } from "@/lib/sync/package-schema"
import { writeAudit } from "@/lib/control/audit"

interface AgentPackageBody {
  siteCode: string
  batchId: string
  sentAt: string
  tables: Array<{
    name: string
    rows: Record<string, unknown>[]
  }>
}

export async function POST(req: NextRequest) {
  const rawBody = await req.text()
  let body: AgentPackageBody
  try {
    body = JSON.parse(rawBody)
  } catch {
    return NextResponse.json({ ok: false, code: "INVALID_JSON" }, { status: 400 })
  }

  // 1. 验签 (R.19A hmac)
  const url = new URL(req.url)
  const auth = verifySiteAgentRequest({
    siteCode: req.headers.get("x-site-code"),
    timestamp: req.headers.get("x-timestamp"),
    nonce: req.headers.get("x-nonce"),
    signature: req.headers.get("x-signature"),
    method: "POST",
    path: url.pathname,
    rawBody,
    payloadSiteCode: body.siteCode,
  })
  if (!auth.ok) {
    return NextResponse.json(
      { ok: false, code: auth.code, message: auth.message },
      { status: 401 }
    )
  }

  // 2. 验 schema: 白名单 6 张
  const accepted: Array<{ table: string; upserted: number; received: number }> = []
  const rejected: Array<{ table: string; error: string }> = []

  for (const tbl of body.tables ?? []) {
    if (!ALLOWED_PACKAGE_TABLES.includes(tbl.name as any)) {
      rejected.push({ table: tbl.name, error: "table not in whitelist" })
      continue
    }
    try {
      const result = await dispatchPackage({
        tableName: tbl.name as any,
        siteCode: body.siteCode,
        records: tbl.rows,
      })
      accepted.push({ table: tbl.name, upserted: result.upserted, received: tbl.rows.length })
    } catch (e) {
      rejected.push({ table: tbl.name, error: e instanceof Error ? e.message : String(e) })
    }
  }

  // 3. 写 sync_package_log + audit (复用 R.2H.3)
  await transaction(async (client) => {
    await client.query(
      `INSERT INTO sync_package_log (site_code, batch_id, package_status, package_type, record_count, created_at)
       VALUES ($1, $2, $3, $4, $5, NOW())
       ON CONFLICT (site_code, batch_id) DO NOTHING`,
      [
        body.siteCode,
        body.batchId,
        rejected.length === 0 ? "success" : "partial",
        "site_agent_push",
        accepted.reduce((s, a) => s + a.received, 0),
      ]
    )
  })

  await writeAudit({
    commandNo: body.batchId,
    action: "site_agent_package",
    targetTable: "site_agent_runtime",
    targetId: body.siteCode,
    before: null,
    after: { accepted, rejected },
    actor: "site-agent",
    siteCode: body.siteCode,
    dryRun: false,
    result: rejected.length === 0 ? "success" : "success", // partial 仍算业务成功 (R.18 不算失败)
    errorMessage: null,
  })

  return NextResponse.json({
    ok: true,
    batchId: body.batchId,
    accepted: accepted.length,
    rejected: rejected.length,
    details: { accepted, rejected },
  })
}
```

// lib/site-agent/sync-pusher.ts
/**
 * Site Agent Sync Pusher (R.19C)
 *
 * 60s 周期: 增量读 source_restore 6 张白名单 → 打 package → POST /api/site-agent/packages.
 *
 * 复用:
 *   - lib/site-agent/pg-reader.ts (Task 2)
 *   - lib/site-agent/hmac.ts (R.19A)
 *   - lib/site-agent/command-queue.ts (Task 1, 用于重试)
 *
 * 错误处理 (R.18 边界):
 *   - 5xx 重试 3 次 (指数退避)
 *   - 4xx 不重试 (认证失败)
 *   - 重试用 CommandQueue 持久化 (Agent 重启可恢复)
 */

import { randomBytes, createHmac } from "node:crypto"
import type { PgReader, SourceTableName } from "./pg-reader"
import { SOURCE_TABLES } from "./pg-reader"
import { getSiteAgentSecret } from "./config"
import type { CommandQueue } from "./command-queue"

export interface SyncPusherConfig {
  siteCode: string
  platformUrl: string
  secret: string
  agentId: string
  maxRetries?: number
  retryBackoffMs?: number
}

export interface PushResult {
  batchId: string
  totalRows: number
  tables: number
  accepted: number
  rejected: number
}

export class SyncPusher {
  private readonly maxRetries: number
  private readonly retryBackoffMs: number
  private readonly watermarks: Map<SourceTableName, string | null> = new Map()

  constructor(
    private readonly config: SyncPusherConfig,
    private readonly pgReader: PgReader,
    private readonly fetchImpl: typeof fetch = fetch,
    private readonly commandQueue?: CommandQueue // 可选, 重试用
  ) {
    this.maxRetries = config.maxRetries ?? 3
    this.retryBackoffMs = config.retryBackoffMs ?? 1000
  }

  async pushOnce(): Promise<PushResult> {
    const batchId = `r19c-${Date.now()}-${randomBytes(3).toString("hex")}`
    const tables: Array<{ name: SourceTableName; rows: Record<string, unknown>[] }> = []
    let totalRows = 0

    for (const tableName of SOURCE_TABLES) {
      const watermark = this.watermarks.get(tableName) ?? null
      const rows = await this.pgReader.readIncremental(tableName, watermark, 500)
      if (rows.length > 0) {
        tables.push({ name: tableName, rows: rows as Record<string, unknown>[] })
        totalRows += rows.length
        // 推进 watermark (用最大 update_dt/create_dt)
        const newWatermark = this.extractMaxTimestamp(rows, tableName)
        if (newWatermark) this.watermarks.set(tableName, newWatermark)
      }
    }

    if (totalRows === 0) {
      return { batchId, totalRows: 0, tables: 0, accepted: 0, rejected: 0 }
    }

    const body = JSON.stringify({
      siteCode: this.config.siteCode,
      batchId,
      sentAt: new Date().toISOString(),
      tables,
    })

    const url = `${this.config.platformUrl}/api/site-agent/packages`
    const headers = this.signRequest(body)

    let lastErr: Error | null = null
    for (let attempt = 0; attempt < this.maxRetries; attempt++) {
      try {
        const res = await this.fetchImpl(url, { method: "POST", headers, body })
        if (res.status >= 200 && res.status < 300) {
          const json = (await res.json()) as { accepted: number; rejected: number }
          return { batchId, totalRows, tables: tables.length, accepted: json.accepted, rejected: json.rejected }
        }
        if (res.status >= 400 && res.status < 500) {
          throw new Error(`push failed: ${res.status} ${await res.text()}`)
        }
        lastErr = new Error(`push failed: ${res.status}`)
      } catch (e) {
        lastErr = e instanceof Error ? e : new Error(String(e))
        if (attempt < this.maxRetries - 1) {
          await new Promise((r) => setTimeout(r, this.retryBackoffMs * (attempt + 1)))
        }
      }
    }
    throw lastErr ?? new Error("push failed after retries")
  }

  private signRequest(body: string): Record<string, string> {
    const timestamp = new Date().toISOString()
    const nonce = randomBytes(16).toString("hex")
    const sigPayload = `${timestamp}.${nonce}.${body}`
    const signature = createHmac("sha256", this.config.secret).update(sigPayload).digest("hex")
    return {
      "content-type": "application/json",
      "x-site-code": this.config.siteCode,
      "x-timestamp": timestamp,
      "x-nonce": nonce,
      "x-signature": signature,
    }
  }

  private extractMaxTimestamp(rows: Record<string, unknown>[], table: SourceTableName): string | null {
    const candidates = ["update_dt", "update_time", "create_dt", "earliest_time"]
    let max: string | null = null
    for (const row of rows) {
      for (const c of candidates) {
        const v = row[c]
        if (typeof v === "string" && (!max || v > max)) max = v
      }
    }
    return max
  }
}

// lib/site-agent/pg-adapter.ts
/**
 * Site Agent PG Action Adapter (R.22)
 *
 * 6 commandType 真实执行 (R.18 capability matrix 决议):
 *   - task_pause: UPDATE status=20 (docx L118-119)
 *   - task_resume: UPDATE status=0 (docx L96-97)
 *   - task_reset: 按 type 区分
 *     - type ∈ {0,2,3,4}: UPDATE status=1, burn_status=0 (R.4 推断, docx 无 reset 官方协议)
 *     - type = 1: unsupported (回迁类 status=1 官方=开始回迁, 不重置)
 *   - task_priority_restore: unsupported (priority 列不存在, R.18)
 *   - inspect_start: unsupported (R.18)
 *   - recovery_start: unsupported (R.18)
 *
 * 边界 (R.18):
 *   - 仅直连 source_restore (测试站点 DB)
 *   - 真实生产 = 站点 app 走 API, R.22 仿真, 不等同
 *   - reset 是 R.4 推断, toast 明标 (前端 R.16 已合规)
 */

import { Pool } from "pg"

export type AdapterStatus = "success" | "failed" | "unsupported"

export interface AdapterResult {
  status: AdapterStatus
  result: {
    affectedRows: number
    targetTable: string
    targetId: string
    before?: unknown
    after?: unknown
    blocker?: string
    reason?: string
  }
  errorMessage: string | null
  durationMs: number
}

const RESET_SUPPORTED_TYPES = new Set([0, 2, 3, 4])

export class PgAdapter {
  private pool: Pool

  constructor(config: { siteDatabaseUrl: string }) {
    this.pool = new Pool({ connectionString: config.siteDatabaseUrl, max: 3, idleTimeoutMillis: 10_000 })
  }

  async execute(commandType: string, targetId: string): Promise<AdapterResult> {
    const start = Date.now()
    const client = await this.pool.connect()
    try {
      switch (commandType) {
        case "task_pause":
          return await this.execTaskPause(client, targetId, start)
        case "task_resume":
          return await this.execTaskResume(client, targetId, start)
        case "task_reset":
          return await this.execTaskReset(client, targetId, start)
        case "task_priority_restore":
        case "inspect_start":
        case "recovery_start":
          return {
            status: "unsupported",
            result: {
              affectedRows: 0,
              targetTable: "tbl_task",
              targetId,
              blocker: "blocked_by_source_schema",
              reason: `commandType=${commandType} 源端缺字段/API, R.18 决议 unsupported`,
            },
            errorMessage: null,
            durationMs: Date.now() - start,
          }
        default:
          return {
            status: "failed",
            result: { affectedRows: 0, targetTable: "tbl_task", targetId },
            errorMessage: `unknown commandType: ${commandType}`,
            durationMs: Date.now() - start,
          }
      }
    } finally {
      client.release()
    }
  }

  private async readTaskType(client: any, targetId: string): Promise<number | null> {
    const id = parseInt(targetId, 10)
    if (Number.isNaN(id)) return null
    const r = await client.query(`SELECT task_type FROM tbl_task WHERE id = $1`, [id])
    return r.rows[0]?.task_type ?? null
  }

  private async execTaskPause(client: any, targetId: string, start: number): Promise<AdapterResult> {
    const id = parseInt(targetId, 10)
    if (Number.isNaN(id)) {
      return { status: "failed", result: { affectedRows: 0, targetTable: "tbl_task", targetId }, errorMessage: "invalid targetId", durationMs: Date.now() - start }
    }
    const r = await client.query(`UPDATE tbl_task SET status = 20, update_dt = NOW() WHERE id = $1`, [id])
    return { status: "success", result: { affectedRows: r.rowCount ?? 0, targetTable: "tbl_task", targetId }, errorMessage: null, durationMs: Date.now() - start }
  }

  private async execTaskResume(client: any, targetId: string, start: number): Promise<AdapterResult> {
    const id = parseInt(targetId, 10)
    if (Number.isNaN(id)) {
      return { status: "failed", result: { affectedRows: 0, targetTable: "tbl_task", targetId }, errorMessage: "invalid targetId", durationMs: Date.now() - start }
    }
    const r = await client.query(`UPDATE tbl_task SET status = 0, update_dt = NOW() WHERE id = $1`, [id])
    return { status: "success", result: { affectedRows: r.rowCount ?? 0, targetTable: "tbl_task", targetId }, errorMessage: null, durationMs: Date.now() - start }
  }

  private async execTaskReset(client: any, targetId: string, start: number): Promise<AdapterResult> {
    const taskType = await this.readTaskType(client, targetId)
    if (taskType === null) {
      return { status: "failed", result: { affectedRows: 0, targetTable: "tbl_task", targetId }, errorMessage: "task not found", durationMs: Date.now() - start }
    }
    if (!RESET_SUPPORTED_TYPES.has(taskType)) {
      return {
        status: "unsupported",
        result: {
          affectedRows: 0,
          targetTable: "tbl_task",
          targetId,
          blocker: "blocked_by_source_schema",
          reason: `reset type=${taskType} 官方无 reset 协议 (docx status=1 在 type=1=开始回迁, type=0/2/3/4=数据准备中, R.4 推断但未官方定义), 需站点提供正式 reset 协议`,
        },
        errorMessage: null,
        durationMs: Date.now() - start,
      }
    }
    const id = parseInt(targetId, 10)
    const r = await client.query(`UPDATE tbl_task SET status = 1, burn_status = 0, update_dt = NOW() WHERE id = $1`, [id])
    return { status: "success", result: { affectedRows: r.rowCount ?? 0, targetTable: "tbl_task", targetId }, errorMessage: null, durationMs: Date.now() - start }
  }

  async close(): Promise<void> {
    await this.pool.end()
  }

// lib/site-agent/ctrl-poller.ts
/**
 * Site Agent Ctrl Poller (R.19D)
 *
 * 5s 周期: 拉 pending → CommandQueue dedupe → ack → PgAdapter 执行 → result 回写.
 *
 * 复用:
 *   - lib/site-agent/command-queue.ts (Task 1, dedupe/state)
 *   - lib/site-agent/pg-adapter.ts (Task 7, 真改 tbl_task)
 *   - lib/site-agent/hmac.ts (R.19A)
 *
 * 状态 (R.18 严口径):
 *   - 仿真 = Agent 拉 + Agent 改, 不等同真站点 app
 *   - reset type=1 返 unsupported, 不造"重置成功"
 */

import { randomBytes, createHmac } from "node:crypto"
import type { CommandQueue } from "./command-queue"
import type { PgAdapter } from "./pg-adapter"

export interface CtrlPollerConfig {
  siteCode: string
  platformUrl: string
  secret: string
  agentId: string
}

export interface PollResult {
  fetched: number
  executed: number
  success: number
  failed: number
  unsupported: number
  deduped: number
}

export class CtrlPoller {
  constructor(
    private readonly config: CtrlPollerConfig,
    private readonly queue: CommandQueue,
    private readonly pgAdapter: PgAdapter,
    private readonly fetchImpl: typeof fetch = fetch
  ) {}

  async pollOnce(): Promise<PollResult> {
    const r: PollResult = { fetched: 0, executed: 0, success: 0, failed: 0, unsupported: 0, deduped: 0 }
    const url = `${this.config.platformUrl}/api/control/commands/pending?siteCode=${encodeURIComponent(this.config.siteCode)}`
    const headers = this.signRequest("")
    const res = await this.fetchImpl(url, { method: "GET", headers })
    if (!res.ok) {
      // 拉取失败不阻塞, 下个 5s 重试
      return r
    }
    const json = (await res.json()) as { ok: boolean; commands: Array<{ id: string; commandNo: string; commandType: string; targetId: string; payload: Record<string, unknown> }> }
    r.fetched = json.commands.length

    for (const cmd of json.commands) {
      // dedupe by commandId via CommandQueue
      const isNew = this.queue.markPending(cmd.id, cmd.commandType, JSON.stringify(cmd.payload))
      if (!isNew) {
        r.deduped++
        continue
      }
      // ack: pending -> pulled
      const ackUrl = `${this.config.platformUrl}/api/control/commands/${cmd.id}/ack`
      const ackHeaders = this.signRequest("")
      const ackRes = await this.fetchImpl(ackUrl, { method: "POST", headers: ackHeaders })
      if (!ackRes.ok) {
        r.failed++
        this.queue.markResult(cmd.id, "failed", "{}", `ack failed: ${ackRes.status}`)
        continue
      }
      this.queue.markPulled(cmd.id)

      // execute via PgAdapter
      r.executed++
      const result = await this.pgAdapter.execute(cmd.commandType, cmd.targetId)
      this.queue.markResult(cmd.id, result.status, JSON.stringify(result.result), result.errorMessage)

      // post result
      const resultUrl = `${this.config.platformUrl}/api/control/commands/${cmd.id}/result`
      const resultBody = JSON.stringify({
        status: result.status,
        result: result.result,
        errorMessage: result.errorMessage,
      })
      const resultHeaders = this.signRequest(resultBody)
      const resultRes = await this.fetchImpl(resultUrl, { method: "POST", headers: resultHeaders, body: resultBody })
      if (!resultRes.ok) {
        r.failed++
        continue
      }

      if (result.status === "success") r.success++
      else if (result.status === "unsupported") r.unsupported++
      else r.failed++
    }
    return r
  }

  private signRequest(body: string): Record<string, string> {
    const timestamp = new Date().toISOString()
    const nonce = randomBytes(16).toString("hex")
    const sigPayload = `${timestamp}.${nonce}.${body}`
    const signature = createHmac("sha256", this.config.secret).update(sigPayload).digest("hex")
    return {
      "content-type": "application/json",
      "x-site-code": this.config.siteCode,
      "x-timestamp": timestamp,
      "x-nonce": nonce,
      "x-signature": signature,
    }
  }
}

import { execSync } from "node:child_process"
import { createHmac, randomBytes } from "node:crypto"

const BASE = process.env.BASE_URL ?? "http://localhost:3000"
const SECRET = process.env.SITE_AGENT_SECRET ?? "test-secret-r19d"
const SITE = process.env.SITE_CODE ?? "SH01"

function check(name: string, ok: boolean, detail?: string) {

function sign(secret = SECRET) {
  const timestamp = new Date().toISOString()
  const nonce = randomBytes(16).toString("hex")
  const sig = createHmac("sha256", secret).update(`${timestamp}.${nonce}.`).digest("hex")
  return { "x-site-code": SITE, "x-timestamp": timestamp, "x-nonce": nonce, "x-signature": sig }
}

async function main() {

  const resultJson = await resultRes.json()
